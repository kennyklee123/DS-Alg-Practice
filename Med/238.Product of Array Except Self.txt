class Solution {
    public int[] productExceptSelf(int[] nums) 
    {
   
        int n = nums.length;
        int[] ans = new int[n];
        
        ans[0] = 1;
        
        //going left to right
        for(int i=1 ;i<n ;i++)
            ans[i] = ans[i-1] * nums[i-1];
        
        //going right to left
        int R = 1;
        
        for(int i=n-1;i>=0;i--)
        {
          ans[i] = R * ans[i];
          R *= nums[i];  
        }
        
        return ans;
    }
}
